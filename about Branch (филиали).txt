// Показать все филиали (branch)
git branch

--------------------------------------

//создат новий филиали (branch)
git branch nameOfBranch

-----------------------------------------

//перемещения на филиалы (branch)
git checkout master // поподёт на филиал master

------------------------------------------------

// перемещат на история с кодом 
git checkout 178fda9

------------------------------------------------

//отправит филиалы(branch) на сервер(github)
git push --set-upstream origin developer	//branch{developer}

-------------------------------------------

// изменить имя текущый филиалы (branch)
git branch -m newName

//создат филиал(branch) и сруазу пойди к ним  
git checkout -b newBranch
